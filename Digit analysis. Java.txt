import java.util.Scanner;


public class DigitAnalyzer {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number for analysis: ");
        int number = Math.abs(sc.nextInt()); // abs handles negative inputs
        int sum = 0;


        // Extract and sum each digit
        while (number > 0) {
            sum += number % 10; // Get the last digit
            number /= 10; // Remove the last digit
        }


        System.out.format("The sum of the digits is: %d%n", sum);
        sc.close();
    }
}
